Widget CodeAdd this to any page where you want this widget

<!-- Use the widget -->
<next-my-invoices></next-my-invoices>

<!-- Listen to events -->
<script>
  const widget = document.querySelector("next-my-invoices");
  widget.addEventListener("invoicesLoaded", (e) => {
    console.log("Invoices loaded:", e.detail);
  });
  widget.addEventListener("invoiceSelected", (e) => {
    console.log("Invoice selected:", e.detail);
  });
  widget.addEventListener("invoiceError", (e) => {
    console.error("Invoice error:", e.detail);
  });
</script>

Universal SetupAdd ONCE to your site (e.g. header or layout template) — all widgets share this

<!-- Load SDK -->
<script type="module"
        src="https://your-host.com/embed-sdk/next-embed.es.js">
</script>

<!-- Initialize with token provider -->
<script type="module">
  import { init } from "https://your-host.com/embed-sdk/next-embed.es.js";

  init({
    tokenProvider: {
      get: async () => {
        const res = await fetch("/api/nw-token", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ wid: "invoices" })
        });
        return (await res.json()).token;
      },
      refresh: async () => {
        const res = await fetch("/api/nw-token?refresh=1", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ wid: "invoices" })
        });
        return (await res.json()).token;
      }
    }
  });
</script>
