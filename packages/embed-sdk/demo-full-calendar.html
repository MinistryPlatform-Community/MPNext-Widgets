Widget Code — add this to any page

<!-- Default: full calendar with all views -->
<next-full-calendar></next-full-calendar>

<!-- Cards view only, no toolbar -->
<next-full-calendar view="cards" show-toolbar="false"></next-full-calendar>

<!-- Calendar view (mini cal + cards) -->
<next-full-calendar view="calendar"></next-full-calendar>

<!-- Listen to events -->
<script>
  const widget = document.querySelector("next-full-calendar");
  widget.addEventListener("calendarLoaded", (e) => {
    console.log("Events loaded:", e.detail.count);
  });
  widget.addEventListener("eventSelected", (e) => {
    console.log("Event clicked:", e.detail);
  });
  widget.addEventListener("viewChanged", (e) => {
    console.log("View changed:", e.detail.view);
  });
  widget.addEventListener("fullCalendarError", (e) => {
    console.error("Calendar error:", e.detail);
  });
</script>
