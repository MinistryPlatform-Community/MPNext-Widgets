Widget CodeAdd this to any page where you want this widget

<!-- Use the widget -->
<next-add-to-calendar event-id="123"></next-add-to-calendar>

<!-- Listen to events -->
<script>
  const widget = document.querySelector("next-add-to-calendar");
  widget.addEventListener("calendarEventLoaded", (e) => {
    console.log("Event loaded:", e.detail);
  });
  widget.addEventListener("addToCalendarError", (e) => {
    console.error("Error:", e.detail);
  });
</script>
