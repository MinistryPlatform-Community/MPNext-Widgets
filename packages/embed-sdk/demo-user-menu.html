Widget Code — add this to any page

<!-- Place in your header -->
<next-user-menu
  mp-base-url="https://my.northwoods.church"
  post-logout-redirect-uri="https://northwoods.church"
></next-user-menu>

<!-- When unauthenticated: renders <mpp-user-login> -->
<!-- When authenticated:    renders avatar + dropdown + account modal -->
<!-- post-logout-redirect-uri defaults to current page if omitted -->

Universal Setup — add ONCE to your site (e.g. header or layout template)

<!-- 1. Load MP Widgets (provides mpp-user-login) -->
<script id="MPWidgets" src="https://my.northwoods.church/widgets/dist/MPWidgets.js"></script>

<!-- 2. Load MPNext Embed SDK -->
<script type="module" src="./dist/next-embed.es.js"></script>
