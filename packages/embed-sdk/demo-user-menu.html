Widget Code — add this to any page

<!-- Place in your header -->
<next-user-menu
  mp-base-url="__MP_BASE_URL__"
  post-logout-redirect-uri="[YourWebUrl]"
></next-user-menu>

<!-- When unauthenticated: renders <mpp-user-login> -->
<!-- When authenticated:    renders avatar + dropdown + account modal -->
<!-- post-logout-redirect-uri defaults to current page if omitted -->

Universal Setup — add ONCE to your site (e.g. header or layout template)

<!-- Load MP Widgets (provides mpp-user-login) -->
<script id="MPWidgets" src="__MP_BASE_URL__/widgets/dist/MPWidgets.js"></script>

<!-- Load MPNext Embed SDK (auto-initializes) -->
<script type="module" src="https://your-host.com/embed-sdk/next-embed.es.js"></script>
